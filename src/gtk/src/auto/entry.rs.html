    fn progress_pulse(&self);

    /// `Window::activate_default` on the window containing the entry, in

    /// ## `setting`

    fn set_activates_default(&self, setting: bool);

    /// Sets the alignment for the contents of the entry. This controls
    /// the horizontal positioning of the contents when the displayed

    ///  Reversed for RTL layouts

    fn set_input_purpose(&self, purpose: InputPurpose);

    fn set_invisible_char(&self, ch: Option<char>);

    /// This is equivalent to:

    /// ## `max`
    /// the maximum length of the entry, or 0 for no maximum.
    ///  (other than the maximum length of entries.) The value passed in will
    ///  be clamped to the range 0-65536.
    fn set_max_length(&self, max: i32);

    ///
    /// Since: 3.12
    ///
    /// ## `n_chars`
    /// the new desired maximum width, in characters

    fn set_max_width_chars(&self, n_chars: i32);

    /// new value
    fn set_overwrite_mode(&self, overwrite: bool);

    /// Sets text to be displayed in `self` when it is empty and unfocused.
    /// This can be used to give a visual hint of the expected contents of
    /// the `Entry`.
    ///
    /// Note that since the placeholder text gets removed when the entry
    /// received focus, using this feature is a bit problematic if the entry
    /// is given the initial focus in a window. Sometimes this can be
    /// worked around by delaying the initial focus setting until the
    /// first key event arrives.
    /// ## `text`
    /// a string to be displayed when `self` is empty an unfocused, or `None`
    fn set_placeholder_text(&self, text: Option<&str>);

    /// Causes the entry’s progress indicator to “fill in” the given

    /// inclusive.
    /// ## `fraction`
    /// fraction of the task that’s been completed
    fn set_progress_fraction(&self, fraction: f64);

    /// Sets the fraction of total entry width to move the progress

    /// fraction between 0.0 and 1.0
    fn set_progress_pulse_step(&self, fraction: f64);

    //#[cfg(feature = "v3_10")]
    //fn set_tabs(&self, tabs: /*Ignored*/&mut pango::TabArray);

    /// Sets the text in the widget to the given

    /// ## `text`
    /// the new text
    fn set_text(&self, text: &str);

    /// Sets whether the contents of the entry are visible or not.
    /// When visibility is set to `false`, characters are displayed
    /// as the invisible char, and will also appear that way when
    /// the text in the entry widget is copied elsewhere.
    ///
    /// By default, GTK+ picks the best invisible character available
    /// in the current font, but it can be changed with
    /// `EntryExt::set_invisible_char`.
    ///
    /// Note that you probably want to set `Entry:input-purpose`
    /// to `InputPurpose::Password` or `InputPurpose::Pin` to
    /// inform input methods about the purpose of this entry,
    /// in addition to setting visibility to `false`.
    /// ## `visible`
    /// `true` if the contents of the entry are displayed
    ///  as plaintext
    fn set_visibility(&self, visible: bool);

    /// Changes the size request of the entry to be about the right size
    /// for `n_chars` characters. Note that it changes the size
    /// request, the size can still be affected by
    /// how you pack the widget into containers. If `n_chars` is -1, the
    /// size reverts to the default entry size.
    /// ## `n_chars`
    /// width in chars
    fn set_width_chars(&self, n_chars: i32);

    /// Converts from a position in the entry’s `pango::Layout` (returned by
    /// `EntryExt::get_layout`) to a position in the entry contents
    /// (returned by `EntryExt::get_text`).
    /// ## `text_index`
    /// byte index into the entry contents
    ///
    /// # Returns
    ///
    /// byte index into the entry layout text
    fn text_index_to_layout_index(&self, text_index: i32) -> i32;

    /// Unsets the invisible char previously set with
    /// `EntryExt::set_invisible_char`. So that the
    /// default invisible char is used again.
    fn unset_invisible_char(&self);
}

Search Tricks

Prefix searches with a type followed by a colon (e.g. fn:) to restrict the search to a given type.

Accepted types are: fn, mod, struct, enum, trait, type, macro, and const.
